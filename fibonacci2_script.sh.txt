#!/bin/bash

# פונקציה לבדוק האם המספר הוא חלק מרצף פיבונאצ'י
function is_fibonacci {
    local n=$1
    local a=0
    local b=1
    local index=1

    # לולאה שמחשבת את רצף פיבונאצ'י עד שמגיעים למספר שווה או גדול מ-n
    while [ $b -lt $n ]; do
        local temp=$b
        b=$((a + b))
        a=$temp
        index=$((index + 1))
    done
    # אם המספר שווה ל-n, מחזירים את האינדקס
    if [ $b -eq $n ]; then
        echo $index
    else
        echo "0"
    fi
}

# קריאה למשתמש להזין מספר
read -p "Enter a number: " number

# קריאה לפונקציה ובדיקת התוצאה
index=$(is_fibonacci $number)

# הדפסת התוצאה בהתאם לבדיקה
if [ $index -ne 0 ]; then
    echo "$number is part of the Fibonacci sequence at index $index"
else
    echo "$number is not part of the Fibonacci sequence"
fi
