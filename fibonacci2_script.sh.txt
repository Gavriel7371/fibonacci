#!/bin/bash

# פונקציה לבדוק האם המספר הוא חלק מסדרת פיבונאצ'י
function is_fibonacci {
    local n=$1
    local a=0
    local b=1

    # לולאה שמחשבת את סדרת פיבונאצ'י עד שמגיעים למספר שווה או גדול מ-n
    while [ $b -lt $n ]; do
        local temp=$b
        b=$((a + b))
        a=$temp
    done

    # אם המספר שווה ל-n, המספר הוא חלק מסדרת פיבונאצ'י
    if [ $b -eq $n ]; then
        echo "yes"
    else
        echo "no"
    fi
}

# קריאה למשתמש להזין מספר
read -p "Enter a number: " number

# קריאה לפונקציה ובדיקת התוצאה
result=$(is_fibonacci $number)

# הדפסת התוצאה בהתאם לבדיקה
if [ "$result" == "yes" ]; then
    echo "$number is part of the Fibonacci sequence."
else
    echo "$number is not part of the Fibonacci sequence."
fi
